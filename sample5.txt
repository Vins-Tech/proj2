pipeline {
    agent any
    stages {
        stage('Clone Code') {
            steps {
                git 'https://github.com/yourusername/docker-jenkins.git'
            }
        }
        stage('Build Docker Image') {
            steps {
                bat 'docker build -t java-app .'
            }
        }
        stage('Run Container') {
            steps {
                bat 'docker stop java-app-container'
                bat 'docker rm java-app-container'
                bat 'docker run -d --name java-app-container java-app'
            }
        }
    }
}
